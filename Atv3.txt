#include "main.h"
#include "stdio.h"


#define cursor_off 0x0C
#define cursor_on 0x0E
#define cursor_blink 0x0F

#define E_0 GPIOC->BRR = (1<<7)
#define E_1 GPIOC->BSRR = (1<<7)

#define RS_0 GPIOA->BRR = (1<<9)
#define RS_1 GPIOA->BSRR = (1<<9)

#define LIGHT_0 GPIOB->BRR = (1)
#define LIGHT_1 GPIOB->BSRR = (1)

#define R1_0 GPIOC->BRR = (1<<2)
#define R1_1 GPIOC->BSRR = (1<<2)

#define R2_0 GPIOC->BRR = (1<<3)
#define R2_1 GPIOC->BSRR = (1<<3)

#define D7_0 GPIOA->BRR = (1<<8)
#define D7_1 GPIOA->BSRR = (1<<8)
#define D6_0 GPIOB->BRR = (1<<14)
#define D6_1 GPIOB->BSRR = (1<<14)
#define D5_0 GPIOB->BRR = (1<<4)
#define D5_1 GPIOB->BSRR = (1<<4)
#define D4_0 GPIOB->BRR = (1<<5)
#define D4_1 GPIOB->BSRR = (1<<5)

ADC_HandleTypeDef hadc1;

UART_HandleTypeDef huart2;

void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_USART2_UART_Init(void);
static void MX_ADC1_Init(void);

void udelay(void);
void delayus(int);
void lcd_wrcom4(uint8_t);
void lcd_wrcom(uint8_t);
void lcd_wrchar(char);
void lcd_init(uint8_t);
void lcd_progchar(void);
void goTo(uint8_t, uint8_t);
void lcd_wrstr(char*);
void lcd_clear(void);
void lcd_backlight(uint8_t);
void rele1_on(void);
void rele1_off(void);
void rele2_on(void);
void rele2_off(void);
void lcd_print(void);

int enter = 8, valor = 0;
int hora = 0, min = 0, seg = 0;
int horaIni = 0;
int duracao = 0;

const uint8_t cronometro[8] = {0b00001110,
							   0b00000100,
							   0b00001110,
							   0b00010101,
							   0b00010111,
							   0b00010001,
							   0b00001110,
							   0b00000000};

const uint8_t relogio[8] = {0b00001110,
							0b00000001,
							0b00000100,
							0b00000100,
							0b00000111,
							0b00010000,
							0b00001110,
							0b00000000};


int __io_putchar(int ch){
	if(ch != '\n') lcd_wrchar(ch);
	return ch;
}

void rele1_on(void){
	R1_1;
	lcd_backlight(1);
	goTo(2,0);
	printf("X\n");
}

void rele1_off(void){
	R1_0;
	lcd_backlight(0);
	goTo(2,0);
	printf(".\n");
}

void rele2_on(void){
	R2_1;
	goTo(2,1);
	printf("X\n");
}

void rele2_off(void){
	R2_0;
	goTo(2,1);
	printf(".\n");
}

void udelay(void){
	int tempo = 7;
	while(tempo--);
}

void delayus(int tempo){
	while(tempo--)udelay();
}

void send_data(uint8_t data){
	if(data & (1 << 0))D4_1; else D4_0;
	if(data & (1 << 1))D5_1; else D5_0;
	if(data & (1 << 2))D6_1; else D6_0;
	if(data & (1 << 3))D7_1; else D7_0;
}

void lcd_wrcom4(uint8_t com4){
	send_data(com4);
	RS_0;
	E_1;
	delayus(5);
	E_0;
	HAL_Delay(5);
}

void lcd_wrcom(uint8_t com4){
	send_data(com4 >> 4);
	RS_0;
	E_1;
	delayus(5);
	E_0;
	delayus(5);
	send_data(com4);
	E_1;
	delayus(5);
	E_0;
	HAL_Delay(5);
}

void lcd_wrchar(char ch){
	send_data(ch >> 4);
	RS_1;
	E_1;
	delayus(5);
	E_0;
	delayus(5);
	send_data(ch);
	RS_1;
	E_1;
	delayus(5);
	E_0;
	HAL_Delay(1);
}

void lcd_init(uint8_t cursor){
	lcd_wrcom4(3);
	lcd_wrcom4(3);
	lcd_wrcom4(3);
	lcd_wrcom4(2);
	lcd_wrcom(0x28);
	lcd_wrcom(cursor);
	lcd_wrcom(0x06);
	lcd_wrcom(0x01);
}

void lcd_progchar(void){
	uint8_t n;
	lcd_wrcom(0x40);
	for(n = 0; n < 8; n++) lcd_wrchar(cronometro[n]);
	for(n = 0; n < 8; n++) lcd_wrchar(relogio[n]);
	lcd_wrcom(0x80);
}

void goTo(uint8_t x, uint8_t y){
	uint8_t com;
	if(y == 0) com = 0x80+x;
	if(y == 1) com = 0xC0+x;
	lcd_wrcom(com);
}

void lcd_wrstr(char *str){
	while(*str != '\0') lcd_wrchar(*str++);
}

void lcd_clear(void){
	goTo(0,0);
	lcd_wrstr("                ");
	goTo(0,1);
	lcd_wrstr("                ");
}

void lcd_backlight(uint8_t backlight){
	if(backlight == 0) LIGHT_0;
	else LIGHT_1;
}

void setHora(){
	uint16_t leitura;
	enter = 8;
	valor = 0;
	while(1){
		HAL_ADC_Start(&hadc1);
		HAL_ADC_PollForConversion(&hadc1, 10);
		leitura = HAL_ADC_GetValue(&hadc1);
		if(leitura < 500){
			if(enter == 8)hora = valor*10;
			if(enter == 9)hora += valor;

			if(enter == 11)min = valor*10;
			if(enter == 12)min += valor;

			if(enter == 14)seg = valor*10;
			if(enter == 15)seg += valor;
			enter++;
			valor = 0;
			if(enter == 10) enter = 11;
			if(enter == 13) enter = 14;
			if(enter == 16) break;
			goTo(enter,1);
			}
			else if(leitura > 500 && leitura < 1000){
				valor++;
				if(enter == 8 && valor > 2) valor = 0;
				if(enter == 9 && (valor > 9 || (hora == 20 && valor > 3))) valor = 0;

				if(((enter == 11) || (enter == 14)) && valor > 5) valor = 0;
				if(((enter == 12) || (enter == 15)) && valor > 9) valor = 0;
				printf("%d\n", valor);
				goTo(enter, 1);

			}
			else if(leitura > 1500 && leitura < 2000){
				valor--;
				if(enter == 8 && valor < 0) valor = 2;
				if(enter == 9 && valor < 0){
					if(hora == 20) valor = 3;
					else valor = 9;
				}

				if(((enter == 11) || (enter == 14)) && valor < 0) valor = 5;
				if(((enter == 12) || (enter == 15)) && valor < 0) valor = 9;

				printf("%d\n", valor);
				goTo(enter, 1);
			}
			HAL_ADC_Stop(&hadc1);
			HAL_Delay(1000);
	}
}

void setHoraInicioDuracao(void){
	uint16_t leitura;
	enter = 8;
	valor = 0;
	while(1){
		HAL_ADC_Start(&hadc1);
		HAL_ADC_PollForConversion(&hadc1, 10);
		leitura = HAL_ADC_GetValue(&hadc1);
		if(leitura < 500){
			if(enter == 8) horaIni = valor*10;
			if(enter == 9) horaIni += valor;
			if(enter == 13) duracao = valor*10;
			if(enter == 14) duracao += valor;
			enter++;
			valor = 0;
			if(enter == 10)	enter = 13;
			if(enter == 15) break;
			goTo(enter, 0);
		}
		else if(leitura > 500 && leitura < 1000){
			valor++;
			if(enter == 8 && valor > 2) valor = 0;
			if(enter == 9 && (valor > 9 || (horaIni == 20 && valor > 3))) valor = 0;

			if((enter == 13) && valor > 5) valor = 0;
			if((enter == 14) && valor > 9) valor = 0;
			printf("%d\n", valor);
			goTo(enter, 0);
			}

		else if(leitura > 1500 && leitura < 2000){
			valor--;
			if(enter == 8 && valor < 0) valor = 2;
			if(enter == 9 && valor < 0){
				if(horaIni == 20) valor = 3;
				else valor = 9;
			}

			if((enter == 13) && valor < 0) valor = 5;
			if((enter == 14) && valor < 0) valor = 9;

			printf("%d\n", valor);
			goTo(enter, 0);
			}
		HAL_ADC_Stop(&hadc1);
		HAL_Delay(1000);
		}
	if(duracao == 0){
		goTo(0,14);
		printf("1\n");
		duracao = 1;
	}
}

void atualizaHora(void){
	seg++;
	if(seg > 59){
		seg = 0;
		min++;
	}
	if(min > 59){
		min = 0;
		hora++;
	}

	if(hora > 23) hora = 0;

	goTo(8,1);
	printf("%02d:%02d:%02d\n", hora, min, seg);
}


int main(void)
{
  HAL_Init();
  SystemClock_Config();

  MX_GPIO_Init();
  MX_USART2_UART_Init();
  MX_ADC1_Init();

  lcd_init(cursor_on);
  lcd_progchar();
  LIGHT_1;
  goTo(0,0);
  printf("L[.]\n");
  goTo(7,0);
  lcd_wrchar(1);
  printf("00\n");
  goTo(12,0);
  lcd_wrchar(0);
  printf("00\n");
  goTo(0,1);
  printf("C[.]\n");
  goTo(8,1);
  printf("00:00:00\n");
  goTo(8,0);
  setHoraInicioDuracao();
  goTo(8,1);
  HAL_Delay(1000);
  setHora();

	while (1){
		if(hora >= 18 && hora <= 22) rele1_on();
		else rele1_off();
		if(horaIni == hora && min <= duracao) rele2_on();
		else rele2_off();

		atualizaHora();
		HAL_Delay(1000);
  }
}
